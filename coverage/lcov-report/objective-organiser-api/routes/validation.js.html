function validateData(objective, schemaType) {
  let isValid = true;
  let errorMessage = null;
  const validatedData = schemaType.validate(objective)[0];
  if (validatedData) {
    isValid = false;
    errorMessage = validatedData.message;
  }
  return { isValid, errorMessage };
}
